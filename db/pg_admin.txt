Что это

	Это графический интерфейс для работы с postgresql.

Установка

	Должен быть установлен питон.

	virtualenv --python=/usr/bin/python3 pgadmin4

	. pgadmin4/bin/activate

	pip3 install https://ftp.postgresql.org/pub/pgadmin/pgadmin4/v3.5/pip/pgadmin4-3.5-py2.py3-none-any.whl

	cd pgadmin4/lib/python3.6/site-packages/pgadmin4

	sudo nano config_local.py

		import os
		DATA_DIR = os.path.realpath(os.path.expanduser(u'~/.pgadmin/'))
		LOG_FILE = os.path.join(DATA_DIR, 'pgadmin4.log')
		SQLITE_PATH = os.path.join(DATA_DIR, 'pgadmin4.db')
		SESSION_DB_PATH = os.path.join(DATA_DIR, 'sessions')
		STORAGE_DIR = os.path.join(DATA_DIR, 'storage')
		SERVER_MODE = False

	python3 pgAdmin4.py

Запуск

	virtualenv

		cd ~/work/virtualenvs

		. pgadmin4/bin/activate

		python3 ~/work/virtualenvs/pgadmin4/lib/python3.6/site-packages/pgadmin4/pgAdmin4.py

Создание сервера

	После запуска переходим на http://127.0.0.1:5050

	Далее правой кнопкой мыши по servers -> create -> server

	На вкладке General вводим имя

	На вкладке Connection вводим host - 127.0.0.1 и password (если есть)
